zTree_v3 相关资源需要你手动复制到`/static/admin`下。
<https://gitee.com/yinqi/Light-Year-Example/tree/master/js/zTree_v3>

```php
//***
use tpext\builder\traits\HasBuilder;

class User extends Controller
{

   public function index()
    {
        $builder = $this->builder($this->pageTitle, $this->indexText);

        $departList = $this->departModel->where('parent_id', 'gt', 0)->select();

        $tree = [
            [
                'id' => '',
                'pId' => 0,
                'name' => '全部',
            ],
        ];

        foreach ($departList as $dep) {
            $tree[] = [
                'id' => $dep['id'],
                'pId' => $dep['parent_id'],
                'name' => $dep['title'],
            ];
        }

        $data = json_encode($tree);
        $script = <<<EOT

        var setting = {
            view: {
              addHoverDom: false,
              removeHoverDom: false,
              selectedMulti: false
            },
            check: {
              enable: false
            },
            data: {
              simpleData: {
                enable: true
              }
            },
            edit: {
              enable: false
            },
            callback: {
                beforeClick: function(treeId, treeNode, clickFlag){
                    $('#form-department').val(treeNode.id).select2();
                    $('#form-submit').trigger('click');
                }
            }
          };
          var zNodes = {$data};

          $(document).ready(function () {
            var treeObj = $.fn.zTree.init($("#treeDemo"), setting, zNodes);
            treeObj.expandAll(true);
          });

EOT;

        $builder->customCss('/static/admin/zTree_v3/css/materialDesignStyle/materialdesign.css');
        $builder->customJs('/static/admin/zTree_v3/js/jquery.ztree.all.min.js');

        $builder->addScript($script);
        
        //样式可选，左边树形占clo-md-1的话有点不够，这里重置一下。把树形宽度再给的大一些的话就没必要。
        $builder->addStyleSheet('.ztree li ul{
            padding:0;
        }
');

        $left = $builder->column(1);

        $left->content()->display('<ul id="treeDemo" class="ztree"></ul>');

        $right = $builder->column(11);

        $this->table = $right->table();
        $this->table->pk($this->dataModel->getPk());
        $this->search = $this->table->getSearch();

        $this->builSearch();
        $this->buildDataList();

        if (request()->isAjax()) {
            return $this->table->partial()->render();
        }

        return $builder->render();
    }
}
```
